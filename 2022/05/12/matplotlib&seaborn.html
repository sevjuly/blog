그래프를 그리기 위해선 matplotlib를 호출해주어야한다.

import matplotlib

import matplotlib.pyplot as plt
plt.plot([1, 2, 4, 9, 5, 3])
plt.show()

fig : 그림을 그리기 위한 도화지라고 이해하면 좋다. 나머지 부분 : 도화지 위에 그려지는 그림, 따로 따로 정의해준다. plt.show() : 화면 출력

fig, ax = plt.subplots()

#fig : 도화지 , 그 뒷 부분 : 그림 , 따로따로 정의해주어야함

# 2. ax 위에 그래프를 그립니다.
ax.plot([1, 2, 4, 9, 5, 3])

# 3. 그래프를 화면에 출력합니다.
plt.show()

import numpy as np
x = np.linspace(-2, 2, 500)  #-2에서 2까지의 범위 사이에서 500개의 부동소수로 구성된 x배열을 만든다.
y = x**2 # x에 거듭제곱한 값을 y배열로 만든다.

plt.plot(x, y)
plt.show()

그래프를 좀 더 꾸며보자. 바로 위의 그래프와 어떤 차이가 있는 지를 살펴보자

plt.plot(x, y)
plt.title("Square function") #그래프의 타이틀을 넣는다
plt.xlabel("x") #x 라벨을 넣고
plt.ylabel("y = x**2") #y 라벨을 넣어준다
plt.grid(True) #plt.grid()는 뒷 배경에 모눈형태를 만들어준다.
plt.show()

object oriented API는 그래프의 각 부분을 객체로 지정하고 그리는 것 대체로 축 이름(label), 범위(limits) 등을 지정하는 명령어는 set_대상(), 거꾸로 그래프에서 설정값을 가져오는 명령어는 get_대상()으로 통일되어 있다.

fig, ax = plt.subplots()

ax.plot(x, y)
ax.set_title("Square function")
ax.set_xlabel("x")
ax.set_ylabel("y = x**2")
ax.grid(True)

plt.show()

재미를 위해서 xkcd 스타일의 그래프를 그려보고 싶다면, with plt.xkcd() 섹션 블록을 활용할 수도 있습니다.

with plt.xkcd():
    plt.plot(x, x**2, px, py, "ro")

    bbox_props = dict(boxstyle="rarrow,pad=0.3", ec="b", lw=2, fc="lightblue")
    plt.text(px-0.2, py, "Beautiful point", bbox=bbox_props, ha="right")

    bbox_props = dict(boxstyle="round4,pad=1,rounding_size=0.2", ec="black", fc="#EEEEFF", lw=5)
    plt.text(0, 1.5, "Square function\n$y = x^2$", fontsize=20, color='black', ha="center", bbox=bbox_props)

    plt.show()

범례를 추가하는 가장 간단한 방법은 모든 선에 라벨을 설정 해 주고, legend 함수를 호출하는 것입니다.

x = np.linspace(-1.4, 1.4, 50)
plt.plot(x, x**2, "r--", label="Square function")
plt.plot(x, x**3, "g-", label="Cube function")
plt.legend(loc="best")
plt.grid(True)
plt.show()

극좌표계 그래프를 그리는 것은 매우 간단합니다. 부분 그래프를 생성할 때 projection 속성을 "polar"로 설정해 주기만 하면 됩니다.

radius = 1
theta = np.linspace(0, 2*np.pi*radius, 1000)

plt.subplot(111, projection='polar')
plt.plot(theta, np.sin(5*theta), "g-")
plt.plot(theta, 0.5*np.cos(20*theta), "b-")
plt.show()
